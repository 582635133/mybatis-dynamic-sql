16  package org.mybatis.dynamic.sql.where.condition;
17  
18  import static org.mybatis.dynamic.sql.util.StringUtilities.spaceAfter;
19  
20  import java.util.Collection;
21  import java.util.function.UnaryOperator;
22  import java.util.stream.Collectors;
23  import java.util.stream.Stream;
24  
25  import org.mybatis.dynamic.sql.AbstractListValueCondition;
26  
27  public class IsIn<T> extends AbstractListValueCondition<T> {
28  
29      protected IsIn(Collection<T> values, UnaryOperator<Stream<T>> valueStreamTransformer) {
30          super(values, valueStreamTransformer);
31      }
32  
33      protected IsIn(Collection<T> values) {
34          super(values);
35      }
36  
37      @Override
38      public String renderCondition(String columnName, Stream<String> placeholders) {
39          return spaceAfter(columnName)
40                  + placeholders.collect(Collectors.joining(",", "in (", ")")); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
41      }
42  
43      /**
44       * This method allows you to modify the condition's values before they are placed into the parameter map.
45       * For example, you could filter nulls, or trim strings, etc. This process will run before final rendering of SQL.
46       * If you filter values out of the stream, then final condition will not reference those values. If you filter all
47       * values out of the stream, then the condition will not render.
48       *   
49       * @param valueStreamTransformer a UnaryOperator that will transform the value stream before
50       *     the values are placed in the parameter map
51       * @return new condition with the specified transformer
52       */
53      public IsIn<T> then(UnaryOperator<Stream<T>> valueStreamTransformer) {
54          return new IsIn<>(values, valueStreamTransformer);
55      }
56  
57      public static <T> IsIn<T> of(Collection<T> values) {
58          return new IsIn<>(values);
59      }
60  }
