16  package org.mybatis.dynamic.sql.select;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  import java.util.Objects;
21  import java.util.function.Function;
22  
23  import org.mybatis.dynamic.sql.BasicColumn;
24  import org.mybatis.dynamic.sql.select.QueryExpressionDSL.FromGatherer;
25  import org.mybatis.dynamic.sql.select.QueryExpressionDSL.FromGathererBuilder;
26  import org.mybatis.dynamic.sql.select.render.SelectStatementProvider;
27  import org.mybatis.dynamic.sql.util.Buildable;
28  
29  /**
30   * Implements a standard SQL dialect for building model classes.
31   * 
32   * @author Jeff Butler
33   *
34   * @param <R> the type of model produced by this builder
35   */
36  public class SelectDSL<R> implements Buildable<R> {
37  
38      private Function<SelectModel, R> adapterFunction;
39      private List<QueryExpressionModel> queryExpressions = new ArrayList<>();    
40      private OrderByModel orderByModel;
41      private Long limit;
42      private Long offset;
43      
44      private SelectDSL(Function<SelectModel, R> adapterFunction) {
45          this.adapterFunction = Objects.requireNonNull(adapterFunction);
46      }
47  
48      private FromGatherer<R> queryExpressionBuilder(BasicColumn...selectList) {
49          return new FromGathererBuilder<R>()
50                  .withSelectDSL(this)
51                  .withSelectList(selectList)
52                  .build();
53      }
54      
55      private FromGatherer<R> distinctQueryExpressionBuilder(BasicColumn...selectList) {
56          return new FromGathererBuilder<R>()
57                  .withSelectDSL(this)
58                  .withSelectList(selectList)
59                  .isDistinct()
60                  .build();
61      }
62      
63      public static FromGatherer<SelectModel> select(BasicColumn...selectList) {
64          return select(Function.identity(), selectList);
65      }
66      
67      public static <R> FromGatherer<R> select(Function<SelectModel, R> adapterFunction,
68              BasicColumn...selectList) {
69          SelectDSL<R> selectModelBuilder = new SelectDSL<>(adapterFunction);
70          return selectModelBuilder.queryExpressionBuilder(selectList);
71      }
72      
73      public static FromGatherer<SelectModel> selectDistinct(BasicColumn...selectList) {
74          return selectDistinct(Function.identity(), selectList);
75      }
76      
77      public static <R> FromGatherer<R> selectDistinct(Function<SelectModel, R> adapterFunction,
78              BasicColumn...selectList) {
79          SelectDSL<R> selectModelBuilder = new SelectDSL<>(adapterFunction);
80          return selectModelBuilder.distinctQueryExpressionBuilder(selectList);
81      }
82      
83      public static <T> FromGatherer<MyBatis3SelectModelAdapter<T>> selectWithMapper(
84              Function<SelectStatementProvider, T> mapperMethod, BasicColumn...selectList) {
85          return select(selectModel -> MyBatis3SelectModelAdapter.of(selectModel, mapperMethod), selectList);
86      }
87      
88      public static <T> FromGatherer<MyBatis3SelectModelAdapter<T>> selectDistinctWithMapper(
89              Function<SelectStatementProvider, T> mapperMethod, BasicColumn...selectList) {
90          return selectDistinct(selectModel -> MyBatis3SelectModelAdapter.of(selectModel, mapperMethod),
91                  selectList);
92      }
93      
94      void addQueryExpression(QueryExpressionModel queryExpression) {
95          queryExpressions.add(queryExpression);
96      }
97      
98      void setOrderByModel(OrderByModel orderByModel) {
99          this.orderByModel = orderByModel;
100     }
101     
102     public LimitFinisher limit(long limit) {
103         this.limit = limit;
104         return new LimitFinisher();
105     }
106 
107     public OffsetFinisher offset(long offset) {
108         this.offset = offset;
109         return new OffsetFinisher();
110     }
111 
112     @Override
113     public R build() {
114         SelectModel selectModel = SelectModel.withQueryExpressions(queryExpressions)
115                 .withOrderByModel(orderByModel)
116                 .withLimit(limit)
117                 .withOffset(offset)
118                 .build();
119         return adapterFunction.apply(selectModel);
120     }
121     
122     public class LimitFinisher implements Buildable<R> {
123         public OffsetFinisher offset(int offset) {
124             return SelectDSL.this.offset(offset);
125         }
126         
127         @Override
128         public R build() {
129             return SelectDSL.this.build();
130         }
131     }
132 
133     public class OffsetFinisher implements Buildable<R> {
134         @Override
135         public R build() {
136             return SelectDSL.this.build();
137         }
138     }
139 }
