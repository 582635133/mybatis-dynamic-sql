40  import org.springframework.jdbc.core.namedparam.SqlParameterSource;
41  import org.springframework.jdbc.datasource.embedded.EmbeddedDatabase;
42  import org.springframework.jdbc.datasource.embedded.EmbeddedDatabaseBuilder;
43  import org.springframework.jdbc.datasource.embedded.EmbeddedDatabaseType;
44  import org.springframework.jdbc.support.GeneratedKeyHolder;
45  import org.springframework.jdbc.support.KeyHolder;
46  
47  import examples.generated.always.GeneratedAlwaysRecord;
48  
49  @RunWith(JUnitPlatform.class)
50  public class SpringTest {
51      private EmbeddedDatabase db;
52      
53      @BeforeEach
54      public void setup() {
55          db = new EmbeddedDatabaseBuilder()
56                  .setType(EmbeddedDatabaseType.HSQL)
57                  .generateUniqueName(true)
58                  .addScript("classpath:/examples/generated/always/CreateGeneratedAlwaysDB.sql")
59                  .build();
60      }
61      
62      @Test
63      public void testSelect() {
64          NamedParameterJdbcTemplate template = new NamedParameterJdbcTemplate(db);
65          
66          SelectStatement selectStatement = selectByExample()
67                  .where(id, isGreaterThan(3))
68                  .orderBy(id.descending())
69                  .build()
70                  .render(RenderingStrategy.SPRING_NAMED_PARAMETER);
71          
72          String expected = "select a.id as A_ID, a.first_name, a.last_name, a.full_name "
73                  + "from GeneratedAlways a "
74                  + "where a.id > :p1 "
75                  + "order by id DESC";
76          
77          assertThat(selectStatement.getSelectStatement()).isEqualTo(expected);
78          
79          List<GeneratedAlwaysRecord> records = template.query(selectStatement.getSelectStatement(), selectStatement.getParameters(),
80                  new RowMapper<GeneratedAlwaysRecord>(){
81                      @Override
82                      public GeneratedAlwaysRecord mapRow(ResultSet rs, int rowNum) throws SQLException {
83                          GeneratedAlwaysRecord record = new GeneratedAlwaysRecord();
84                          record.setId(rs.getInt(1));
85                          record.setFirstName(rs.getString(2));
86                          record.setLastName(rs.getString(3));
87                          return record;
88                      }
89                  });
90          
91          SoftAssertions.assertSoftly(softly -> {
92              softly.assertThat(records.size()).isEqualTo(3);
93              softly.assertThat(records.get(0).getId()).isEqualTo(6);
94              softly.assertThat(records.get(1).getId()).isEqualTo(5);
95              softly.assertThat(records.get(2).getId()).isEqualTo(4);
96          });
97      }
98      
99      @Test
100     public void testInsert() {
101         NamedParameterJdbcTemplate template = new NamedParameterJdbcTemplate(db);
102         
103         GeneratedAlwaysRecord record = new GeneratedAlwaysRecord();
104         record.setId(100);
105         record.setFirstName("Bob");
106         record.setLastName("Jones");
107         
108         InsertStatement<GeneratedAlwaysRecord> insertStatement = buildInsert(record);
109         
110         SqlParameterSource ps = new BeanPropertySqlParameterSource(record);
111         KeyHolder kh = new GeneratedKeyHolder();
112         
113         int rows = template.update(insertStatement.getInsertStatement(), ps, kh);
114         
115         SoftAssertions.assertSoftly(softly -> {
116             softly.assertThat(rows).isEqualTo(1);
117             softly.assertThat(kh.getKeys().get("FULL_NAME")).isEqualTo("Bob Jones");
118         });
119     }
120     
121     @AfterEach
122     public void teardown() {
123         db.shutdown();
124     }
125 }
